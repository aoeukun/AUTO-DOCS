    """
This script automates the generation of Python code documentation using the Google Gemini API.

It processes Python files in a specified source directory, generates documentation for each file using the Gemini model,
and saves the documentation as Markdown files in a designated output directory.
Furthermore, it updates the `mkdocs.yml` configuration file to include the generated documentation in the MkDocs navigation,
making the documentation easily accessible within a MkDocs website.
"""
import os
import json
from pathlib import Path
import yaml
from dotenv import load_dotenv
import google.generativeai as genai

load_dotenv()
API_KEY = os.getenv("GOOGLE_API_KEY") # Load the Google API key from environment variables.

# Gemini setup
genai.configure(api_key=API_KEY) # Configure the Gemini API client with the API key.

MODEL_NAME = "models/gemini-2.0-flash-thinking-exp-01-21" # Specify the Gemini model to be used for documentation generation.

SYSTEM_PROMPT = """Act as a Python code documentation assistant. Your task is to add comprehensive documentation to the provided Python code snippet, making it clear, understandable, and maintainable.

**Instructions:**

1.  **Analyze the Code:** Understand the purpose and logic of the provided Python code.
2.  **Add Docstrings:**
    * Include a **module-level docstring** at the very beginning of the script explaining its overall purpose and functionality.
    * Add **function/method/class docstrings** immediately following their definition lines (`def` or `class`).
    * Follow a clear and standard convention, preferably **Google style**:
        * Start with a concise one-line summary (using the imperative mood, e.g., "Calculate..." not "Calculates..."). End with a period.
        * Include a blank line after the summary if more detail follows.
        * Add further elaboration on the object's purpose or logic if necessary.
        * Use an `Args:` section to detail each parameter (`parameter_name (type): Description of the parameter.`).
        * Use a `Returns:` section to detail the return value (`type: Description of the return value.`). If the function doesn't return anything explicitly (returns `None`), you can state that or omit the section.
        * Use a `Raises:` section (if applicable) to detail any specific exceptions the code might intentionally raise (`ExceptionType: Condition under which it's raised.`).
3.  **Add Inline Comments:** Insert inline comments (`#`) judiciously to clarify specific lines or blocks of code that involve complex logic, non-obvious operations, or important algorithmic steps. Avoid commenting on obvious code.
4.  **Maintain Code Integrity:** Do not change the original code's logic or functionality. Only add documentation elements (docstrings and comments).
5.  **Output Format:** Return the *complete* Python code, including the original logic, with all the added docstrings and relevant inline comments integrated directly into the code. Ensure the output is presented as a single, well-formatted Python code block.

**Python Code to Document:**
"""

def call_gemini(code: str) -> str:
    """Calls the Google Gemini model to generate documentation for the given Python code.

    This function takes a string of Python code as input and sends it to the Gemini model
    along with a system prompt instructing it to act as a Python documentation assistant.
    It then returns the generated documentation text from the Gemini model's response.

    Args:
        code (str): The Python code snippet to be documented.

    Returns:
        str: The documentation generated by the Gemini model as a string.
    """
    model = genai.GenerativeModel(MODEL_NAME) # Initialize the Gemini Generative Model.
    response = model.generate_content([
        {"role": "user", "parts": [{"text": f"{SYSTEM_PROMPT}\n```python\n{code}\n```"}]} # Construct the prompt with system instructions and the code.
    ])
    return response.text # Return the text content of the Gemini response, which is the generated documentation.

def generate_docs(code_path: Path, output_path: Path):
    """Generates documentation for a Python file and saves it to a Markdown file.

    Reads the Python code from the specified `code_path`, calls the `call_gemini` function
    to generate documentation, and then writes the generated documentation to a Markdown file
    at the `output_path`. The output Markdown file includes a header with the name of the
    original Python file.

    Args:
        code_path (Path): Path to the Python file for which documentation needs to be generated.
        output_path (Path): Path to save the generated Markdown documentation file.
    """
    code = code_path.read_text(encoding="utf-8") # Read the content of the Python file.
    documentation = call_gemini(code) # Generate documentation using the Gemini model.
    output_path.parent.mkdir(parents=True, exist_ok=True) # Ensure the output directory exists, creating parent directories if needed.
    output_path.write_text(f"# Documentation for `{code_path.name}`\n\n{documentation}", encoding="utf-8") # Write the documentation to the output Markdown file, including a header.

def update_mkdocs_yml():
    """Updates the mkdocs.yml file to include the generated documentation in the navigation.

    This function reads the `mkdocs.yml` configuration file, finds the 'nav' section,
    and adds a new section named 'Generated Docs' containing links to all generated
    Markdown documentation files found in the 'docs/generated' directory.
    It ensures that if a 'Generated Docs' section already exists, it is replaced with the new one.
    Finally, it writes the updated configuration back to `mkdocs.yml`.
    """
    GENERATED_DIR = Path("docs/generated") # Define the directory where generated documentation files are stored.
    MKDOCS_YML_PATH = Path("mkdocs.yml") # Define the path to the mkdocs.yml configuration file.

    with open(MKDOCS_YML_PATH, "r", encoding="utf-8") as f: # Open mkdocs.yml in read mode.
        config = yaml.safe_load(f) # Load the YAML configuration from mkdocs.yml.

    files = [f for f in GENERATED_DIR.iterdir() if f.suffix == ".md"] # Get a list of all Markdown files in the generated documentation directory.
    files.sort() # Sort the list of files alphabetically.

    new_nav = [] # Initialize a new navigation list.
    for item in config.get("nav", []): # Iterate through the existing navigation items in mkdocs.yml.
        if isinstance(item, dict) and "Generated Docs" in item: # Check if a 'Generated Docs' section already exists and skip it for replacement.
            continue
        new_nav.append(item) # Append existing navigation items to the new navigation list.

    generated_section = {"Generated Docs": []} # Create a new section for generated documentation.
    for file in files: # Iterate through the generated documentation files.
        title = file.stem.replace("_", " ").title() # Create a title for the documentation link from the filename.
        generated_section["Generated Docs"].append({title: f"generated/{file.name}"}) # Add a navigation link to the generated documentation file.

    new_nav.append(generated_section) # Append the new 'Generated Docs' section to the navigation list.
    config["nav"] = new_nav # Update the 'nav' section in the configuration with the new navigation list.

    with open(MKDOCS_YML_PATH, "w", encoding="utf-8") as f: # Open mkdocs.yml in write mode.
        yaml.dump(config, f, sort_keys=False) # Write the updated YAML configuration back to mkdocs.yml, preserving order.

    print("✅ mkdocs.yml updated successfully.") # Print a success message to the console.

def main():
    """Main function to orchestrate the documentation generation and mkdocs update process.

    This function sets up the source and output directories, iterates through all Python files
    in the source directory (recursively), generates documentation for each file using `generate_docs`,
    and finally calls `update_mkdocs_yml` to update the MkDocs navigation.
    """
    src_dir = Path("src") # Define the source directory where Python files are located.
    out_dir = Path("docs/generated") # Define the output directory for generated documentation files.

    for file in src_dir.rglob("*.py"): # Recursively iterate through all Python files in the source directory.
        generate_docs(file, out_dir / f"{file.stem}.md") # Generate documentation for each Python file and save it in the output directory.

    update_mkdocs_yml() # Update mkdocs.yml to include the generated documentation in navigation.
    print("🎉 Documentation generated and mkdocs navigation updated.") # Print a completion message.

if __name__ == "__main__":
    main() # Execute the main function when the script is run directly.
