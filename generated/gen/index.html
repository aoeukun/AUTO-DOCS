    """
This script generates documentation for Python code files using the Google Gemini API
and updates the navigation configuration of an mkdocs.yml file to include the
generated documentation pages.

It reads source files from a specified directory, calls the Gemini API to
generate markdown documentation for each file, saves the documentation to
markdown files in an output directory, and then modifies the mkdocs.yml
file to add a 'Generated Docs' section in the navigation.
"""

import os
import json
from pathlib import Path
import yaml
from dotenv import load_dotenv
import google.generativeai as genai

# Load environment variables from a .env file
load_dotenv()

# Retrieve the Google API key from environment variables
API_KEY = os.getenv("GOOGLE_API_KEY")

# Gemini API configuration
genai.configure(api_key=API_KEY)

# Specify the name of the Gemini model to use for documentation generation
MODEL_NAME = "models/gemini-2.0-flash-thinking-exp-01-21"

# System prompt provided to the Gemini model to instruct it on documentation generation
SYSTEM_PROMPT = """Act as a Python code documentation assistant. Your task is to add comprehensive documentation to the provided Python code snippet, making it clear, understandable, and maintainable.

**Instructions:**

1.  **Analyze the Code:** Understand the purpose and logic of the provided Python code.
2.  **Add Docstrings:**
    * Include a **module-level docstring** at the very beginning of the script explaining its overall purpose and functionality.
    * Add **function/method/class docstrings** immediately following their definition lines (`def` or `class`).
    * Follow a clear and standard convention, preferably **Google style**:
        * Start with a concise one-line summary (using the imperative mood, e.g., "Calculate..." not "Calculates..."). End with a period.
        * Include a blank line after the summary if more detail follows.
        * Add further elaboration on the object's purpose or logic if necessary.
        * Use an `Args:` section to detail each parameter (`parameter_name (type): Description of the parameter.`).
        * Use a `Returns:` section to detail the return value (`type: Description of the return value.`). If the function doesn't return anything explicitly (returns `None`), you can state that or omit the section.
        * Use a `Raises:` section (if applicable) to detail any specific exceptions the code might intentionally raise (`ExceptionType: Condition under which it's raised.`).
3.  **Add Inline Comments:** Insert inline comments (`#`) judiciously to clarify specific lines or blocks of code that involve complex logic, non-obvious operations, or important algorithmic steps. Avoid commenting on obvious code.
4.  **Maintain Code Integrity:** Do not change the original code's logic or functionality. Only add documentation elements (docstrings and comments).
5.  **Output Format:** Return the *complete* Python code, including the original logic, with all the added docstrings and relevant inline comments integrated directly into the code. Ensure the output is presented as a single, well-formatted Python code block.

**Python Code to Document:**
"""

def call_gemini(code: str) -> str:
    """
    Calls the configured Gemini model to generate documentation for a given Python code snippet.

    Args:
        code (str): The Python code snippet for which documentation is to be generated.

    Returns:
        str: The text response from the Gemini model, expected to be the generated documentation.
    """
    # Instantiate the generative model
    model = genai.GenerativeModel(MODEL_NAME)
    # Send the system prompt and the user's code wrapped in a markdown code block
    response = model.generate_content([
        {"role": "user", "parts": [{"text": f"{SYSTEM_PROMPT}\n```python\n{code}\n```"}]}
    ])
    return response.text

def generate_docs(code_path: Path, output_path: Path):
    """
    Reads a Python code file, generates documentation using `call_gemini`,
    and writes the documentation to a markdown file.

    Args:
        code_path (Path): The path to the input Python code file.
        output_path (Path): The path where the output markdown documentation
                            file should be saved.
    """
    # Read the content of the Python code file
    code = code_path.read_text(encoding="utf-8")
    # Generate documentation using the Gemini model
    documentation = call_gemini(code)

    # Ensure the parent directory for the output file exists
    output_path.parent.mkdir(parents=True, exist_ok=True)
    # Write the generated documentation to the output markdown file
    # Adds a header indicating the source file
    output_path.write_text(f"# Documentation for `{code_path.name}`\n\n{documentation}", encoding="utf-8")

def update_mkdocs_yml():
    """
    Updates the mkdocs.yml file's navigation section to include links to
    the generated documentation markdown files.

    It reads the existing mkdocs configuration, finds all generated markdown
    files in a specific directory, builds a new 'Generated Docs' navigation
    item, removes any existing 'Generated Docs' section, and adds the new one.
    """
    # Define the directory where generated markdown files are stored
    GENERATED_DIR = Path("docs/generated")
    # Define the path to the mkdocs configuration file
    MKDOCS_YML_PATH = Path("mkdocs.yml")

    # Read the existing mkdocs.yml configuration
    with open(MKDOCS_YML_PATH, "r", encoding="utf-8") as f:
        config = yaml.safe_load(f)

    # Find all markdown files in the generated documentation directory
    files = [f for f in GENERATED_DIR.iterdir() if f.suffix == ".md"]
    # Sort the files alphabetically for consistent navigation order
    files.sort()

    # Build the new navigation list, removing any previous 'Generated Docs' section
    new_nav = []
    for item in config.get("nav", []):
        # Check if the item is a dictionary and contains the 'Generated Docs' key
        if isinstance(item, dict) and "Generated Docs" in item:
            # Skip this item (the old generated section)
            continue
        # Keep all other navigation items
        new_nav.append(item)

    # Create the structure for the new 'Generated Docs' navigation section
    generated_section = {"Generated Docs": []}
    # Add each generated markdown file to the 'Generated Docs' section
    for file in files:
        # Create a title from the file name (e.g., 'my_module.md' -> 'My Module')
        title = file.stem.replace("_", " ").title()
        # Add the file to the navigation list within the generated section
        generated_section["Generated Docs"].append({title: f"generated/{file.name}"})

    # Append the newly created 'Generated Docs' section to the navigation
    new_nav.append(generated_section)
    # Update the navigation configuration in the main config dictionary
    config["nav"] = new_nav

    # Write the updated configuration back to mkdocs.yml
    # sort_keys=False maintains the original order of other nav items
    with open(MKDOCS_YML_PATH, "w", encoding="utf-8") as f:
        yaml.dump(config, f, sort_keys=False)

    # Print a success message
    print("✅ mkdocs.yml updated successfully.")

def main():
    """
    Main function to orchestrate the documentation generation and mkdocs update process.

    Iterates through all Python files in the 'src' directory (including subdirectories),
    generates documentation for each, saves it to the 'docs/generated' directory,
    and finally updates the mkdocs navigation.
    """
    # Define the source directory containing Python files
    src_dir = Path("src")
    # Define the output directory for generated markdown files
    out_dir = Path("docs/generated")

    # Iterate through all files ending with .py in the source directory and its subdirectories
    for file in src_dir.rglob("*.py"):
        # Generate documentation for the current Python file
        # The output path is constructed based on the source file name (without extension)
        generate_docs(file, out_dir / f"{file.stem}.md")

    # After generating documentation for all files, update the mkdocs.yml file
    update_mkdocs_yml()

    # Print a final completion message
    print("🎉 Documentation generated and mkdocs navigation updated.")

# Entry point of the script
if __name__ == "__main__":
    main()
